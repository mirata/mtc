ACTOR MagnumPickup : CustomInventory replaces pistol
{
	Radius 16
	Height 16
	Inventory.PickupMessage ".44 magnum mega class"
        scale 0.5
  //YScale 0.5
	States
	{
	Spawn:
		MAGN H -1
		Loop
	Pickup:
		TNT1 A 0 A_JumpIfInventory("MagnumCheck",2,"AlreadyHasMagnums") // Skip ahead if already has 2 Magnums
		TNT1 A 0 A_JumpIfInventory("MagnumCheck",1,"GiveMagnums") // Check for player having only 1 magnum, if so, give him "Magnums"
		TNT1 A 0 A_GiveInventory("MagnumCheck",1) // Now has 1 magnum
		TNT1 A 0 A_GiveInventory("Magnum",1) // Now has Magnum Weapon
		TNT1 A 0 A_GiveInventory("MagnumClip",8) // Load the magnum
		Stop
		TNT1 A 0
		Fail
	GiveMagnums:
		TNT1 A 0 A_GiveInventory("MagnumCheck",1) // Now has 2 magnums
		TNT1 A 0 A_GiveInventory("Magnums",1) // Now has Magnums Weapon
		TNT1 A 0 A_GiveInventory("MagnumClip3",8) // Load the left hand magnum
		TNT1 A 0 A_GiveInventory("MagnumClip2",8) // Load half of Double Magnums
		Stop
	AlreadyHasMagnums:
		TNT1 A 0 A_GiveInventory("MagnumAmmo",1) // Give a clip
		Stop
	}
}
