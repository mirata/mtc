ACTOR MagnumPickup : CustomInventory replaces pistol
{
	Radius 16
	Height 16
	Inventory.PickupMessage ".44 magnum mega class"
        scale .5
  //YScale 0.5
	States
	{
	Spawn:
		MAGN H -1
		Loop
	Pickup:
		TNT1 A 0 A_JumpIfInventory("MagnumCheck",2,"AlreadyHasMagnums") // Skip ahead if already has 2 Magnums
		TNT1 A 0 A_JumpIfInventory("MagnumCheck",1,"GiveMagnums") // Check for player having only 1 magnum, if so, give him "Magnums"
		TNT1 A 0 A_GiveInventory("MagnumCheck",1) // Now has 1 magnum
		TNT1 A 0 A_GiveInventory("Magnum",1) // Now has Magnum Weapon
		TNT1 A 0 A_GiveInventory("MagnumClip",8) // Load the magnum
		Stop
		TNT1 A 0
		Fail
	GiveMagnums:
		TNT1 A 0 A_GiveInventory("MagnumCheck",1) // Now has 2 magnums
		TNT1 A 0 A_GiveInventory("Magnums",1) // Now has Magnums Weapon
		TNT1 A 0 A_GiveInventory("MagnumClip3",8) // Load the left hand magnum
		TNT1 A 0 A_GiveInventory("MagnumClip2",8) // Load half of Double Magnums
		Stop
	AlreadyHasMagnums:
		TNT1 A 0 A_GiveInventory("MagnumAmmo",1) // Give a clip
		Stop
	}
}

actor FusionPistol : Weapon
{
  Game Doom
  SpawnID 30
  Weapon.SelectionOrder 100
  Weapon.AmmoUse 1
  Weapon.AmmoGive 20
  Weapon.AmmoType "FusionClip"
  Weapon.AmmoUse2 1
  Weapon.AmmoType2 "FusionAmmo"
  //Weapon.WeaponScaleY 1
  Inventory.PickupMessage "Zeus-Class Fusion Pistol"
  scale .5
  //YScale 0.5
  +Ammo_optional
  +Alt_Ammo_optional
  +WEAPON.NOALERT
  +NOAUTOAIM
  weapon.yadjust 12
  Weapon.PreferredSkin "PlayerFusion"
  States
  {
  Ready:
    FUSI A 1 A_WeaponReady
    Loop
  Deselect:
    FUSI A 1 A_Lower
    Loop
  Select:
    FUSI A 1 A_Raise
    Loop
  Fire:
    FUSI A 0 A_JumpIfNoammo("Dry")
    FUSI A 0 A_JumpIfInventory("FusionCharge",8,"Fire2")
    FUSI A 1 A_GiveInventory("FusionCharge",1)
    FUSI A 0 A_ReFire
    FUSI B 0 bright NoiseAlert(0,0)
    FUSI A 0 A_GunFlash
    FUSI A 0 A_FireCustomMissile("FusionLight",0,0,0,0,0)
    FUSI A 0 A_Light1
    FUSI B 3 bright A_FireCustomMissile("FusionBall",random(-1,1),1,0,0,0,random(-1,1))
    FUSI A 0 A_Light0
    FUSI A 0 A_Recoil(0.25)
    FUSI C 2 A_TakeInventory("FusionCharge",20)
    FUSI D 3
    FUSI A 0 A_JumpIfNoammo("Dry")
    Goto Ready
    //FUSI A 0 ACS_Execute(780,0,0,0,0)
    //Goto Ready
  Fire2:
    FUSI A 20 A_StartSound("FUSION3", CHAN_WEAPON)
    FUSI B 0 bright NoiseAlert(0,0)
    FUSI A 0 A_GunFlash
    FUSI A 0 A_Light2
    FUSI A 0 A_FireCustomMissile("FusionLight",0,0,0,0,0)
    FUSI B 0 bright A_FireCustomMissile("PoweredFusionBall",0,1,0,0,0,0)
    FUSI B 3 bright A_takeInventory ("FusionClip",4)
    FUSI A 0 A_Light0
    FUSI A 0 A_Recoil(0.75)
    FUSI C 2 A_TakeInventory("Fusioncharge",20)
    FUSI D 3
    FUSI A 0 A_JumpIfNoammo("Dry")
    Goto Ready
    //FUSI A 0 ACS_Execute(780,0,0,0,0)
    //Goto Ready
  Dry:
    FUSI A 0 A_JumpIfInventory("FusionAmmo",1,"Reload")
    FUSI A 1
    //FUSI A 0 ACS_Execute(780,0,0,0,0)
    Goto Ready
  Reload:
    FUSI A 0 A_JumpIfinventory("FusionClip",20,"Ready")
    FUSI AAAAAAAAAAAA 2 A_Lower
    FUSI A 20
    FUSI A 0 A_TakeInventory("FusionAmmo",1)
    FUSI A 0 A_GiveInventory("FusionClip",20)
    FUSI AAAAAAAAAAAA 2 A_Raise
    goto ready
  Spawn:
    FUSI E -1
    Stop
  }
}
